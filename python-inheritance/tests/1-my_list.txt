#!/usr/bin/python3
from 1-my_list import MyList

# Test: check instantiation
>>> my_list = MyList()
>>> print(my_list)
[]

# Test: check inherits from list
>>> isinstance(my_list, list)
True

# Test: check __str__
>>> print(MyList())
[]

# Test: check append()
>>> my_list = MyList()
>>> my_list.append(10)
>>> print(my_list)
[10]

# Test: check print_sorted() with sorted append
>>> my_list = MyList([1, 2, 3])
>>> my_list.print_sorted()
[1, 2, 3]

# Test: check print_sorted() with not sorted append
>>> my_list = MyList([3, 1, 2])
>>> my_list.print_sorted()
[1, 2, 3]

# Test: check print_sorted() with negative number
>>> my_list = MyList([-3, 5, 1])
>>> my_list.print_sorted()
[-3, 1, 5]

# Test: check print_sorted() with empty list
>>> MyList().print_sorted()
[]

# Test: check print_sorted() returns a new list
>>> my_list = MyList([3, 2, 1])
>>> my_list.print_sorted()
[1, 2, 3]
>>> print(my_list)
[3, 2, 1]
